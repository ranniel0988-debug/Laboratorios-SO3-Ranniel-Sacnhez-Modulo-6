sudo apt update
sudo apt install snort -y
sudo nano /etc/snort/rules/local.rules
#coloca las reglas snort
alert icmp any any -> $HOME_NET any (msg:"ALERTA: Trafico ICMP Detectado"; sid:1000001; rev:1;)

alert tcp any any -> $HOME_NET 21 (msg:"ALERTA: Intento FTP Puerto 21"; sid:1000002; rev:1;)

alert tcp any any -> $HOME_NET 22 (msg:"ALERTA: Acceso SSH Puerto 22"; sid:1000003; rev:1;)

alert tcp any any -> $HOME_NET 80 (msg:"ALERTA: Trafico HTTP Puerto 80"; sid:1000004; rev:1;)



sudo nano /etc/snort/snort.lua
#aqui external net debe estar en any
y en la parte donde dice ips agregar la siguiente linea
include = '/etc/snort/rules/local.rules',


sudo systemctl status snort.service
sudo nano /etc/snort/snort.lua
sudo snort -c /etc/snort/snort.lua -T
clear
sudo snort -c /etc/snort/snort.lua -i ens33 -A alert_fast
ip a
sudo snort -c /etc/snort/snort.lua -i ens33 -A alert_fast
